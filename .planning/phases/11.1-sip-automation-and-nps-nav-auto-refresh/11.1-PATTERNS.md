```swift
private func performFetch(context: ModelContext, todayIST: Date) async {
    defer { isFetching = false }
    do {
        let (data, _) = try await URLSession.shared.data(from: Self.npsLatestURL)
        let schemes = parseNPSLatest(data)               // was parseNAVAll(text)

        var newCache: [String: NPSScheme] = [:]
        for scheme in schemes { newCache[scheme.code] = scheme }
        cachedSchemes = newCache

        // Update Asset.currentNAV + navAsOfDate for matching npsSchemeCode (was amfiSchemeCode)
        let assets = try context.fetch(FetchDescriptor<Asset>())
        var didChange = false
        for asset in assets {
            guard let code = asset.npsSchemeCode,          // was asset.amfiSchemeCode
                  let scheme = newCache[code] else { continue }
            asset.currentNAV = scheme.nav
            asset.navAsOfDate = todayIST                   // npsnav.in /latest-min has no per-entry date
            didChange = true
        }
        if didChange { try context.save() }

        UserDefaults.standard.set(todayIST, forKey: Self.lastFetchKey)
    } catch {
        print("[NPSNavService] fetch/parse failed: \(error)")  // silent-fail — staleness badge handles UX
```

**Parser — NEW (not copied from AMFINavService):**

The AMFINavService parser handles `\n`-separated semicolon text. NPSNavService parses a JSON array-of-2-element-arrays where `nav` is a JSON Number (not String). Decode via a custom `Decodable`:
```swift
// npsnav.in /api/latest-min response: [[code, nav], ...] — nav is JSON Number
private struct NPSEntry: Decodable {
    let code: String
    let nav: Double   // intermediate; convert to Decimal immediately — Pitfall 17
    init(from decoder: Decoder) throws {
        var c = try decoder.unkeyedContainer()
        code = try c.decode(String.self)
        nav  = try c.decode(Double.self)
    }
}

func parseNPSLatest(_ data: Data) -> [NPSScheme] {
    guard let entries = try? JSONDecoder().decode([NPSEntry].self, from: data) else { return [] }
    return entries.compactMap { entry in
        // Safe string-based conversion — Decimal(floatLiteral:) is lossy (Pitfall 17)
        guard let nav = Decimal(string: String(format: "%.4f", entry.nav)) else { return nil }
        return NPSScheme(code: entry.code, nav: nav)
    }
}
```

**`NPSScheme` struct — parallel to `AMFIScheme`:**
```swift
// AMFIScheme (analog, lines 9-21 of AMFINavService.swift):

struct NPSScheme: Identifiable {
    let code: String
```

## Critical Cross-Cutting Warnings

1. **STAB-08 typealias commit:** All 10 `@Model` typealiases in `Persistence/Models/` must flip from `SchemaV7.X` to `SchemaV8.X` in the same commit that appends `SchemaV8.self` to `AppMigrationPlan.schemas`. Never flip fewer than all 10.

2. **xcodeproj 4-edit rule (memory `[[xcodeproj-explicit-file-refs]]`):** Every new `.swift` file requires 4 manual edits to `project.pbxproj`: PBXFileReference, PBXBuildFile, PBXSourcesBuildPhase (app target), PBXSourcesBuildPhase (test target). Affects: NPSNavService.swift, SIPAccrualService.swift, NPSSchemePickerView.swift, SIPSetupView.swift, ContributionLogView.swift, ReconcileView.swift, SchemaV8.swift, SIP.swift, SIPAmountChange.swift, Contribution.swift = **10 new files × 4 edits = 40 pbxproj edits**.

3. **Date formatter isolation:** `AMFINavService.navDateFormatter` uses `"dd-MMM-yyyy"` (e.g. `"10-Jun-2026"`). npsnav.in and mfapi.in history use `"dd-MM-yyyy"` (e.g. `"10-06-2026"`). Never share these formatters. Define `historyDateFormatter` as a separate static constant (pattern shown under NPSNavService section above).

4. **Decimal from JSON Number:** npsnav.in returns NAV as a bare JSON Number. Decode as `Double` intermediate, then `Decimal(string: String(format: "%.4f", doubleValue))`. Never `Decimal(exactly: doubleValue)` or `Decimal(floatLiteral: doubleValue)`.

5. **lastAccruedDate advances after save:** In `SIPAccrualService`, advance `sip.lastAccruedDate` only after `try context.save()` succeeds — not before. Failure leaves the cursor unchanged for safe re-entry on next launch.
